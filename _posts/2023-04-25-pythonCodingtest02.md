def main():
    N, M, K = map(int, input().split())  # 5,8,3
    data = list(map(int,input().split())) # [2,4,5,4,6]

    data.sort() # [2,4,5,4,6]
    most_one = data[N-1]
    most_second = data[N-2]

    count = M//(K+1) * K + M % (K+1)

    result = 0
    result += count * most_one
    result += (M-count) * most_second

    print(result)

if __name__ == "__main__":
    main()
